    function receiveMessage(event)
    {
        if (event.origin !== "https://app.recruiterbox.com")
            return;

        var data = event.data;

        if (data === 'FINISHED_LOADING_LINKEDIN_JS') {
            $('.js-apply-with-linkedin').removeClass('hide');
        } else if (data === 'PROFILE_DATA_BEING_FETCHED') {
            show_application_form();
            show_loading_overlay();
        } else if (data.message == 'ERROR_WHILE_FETCHING_PROFILE_DATA') {
            hide_loading_overlay();
            show_application_form();
            _RB.logToRollbar(data.error);
        } else if (data.message == 'LINKEDIN_PROFILE_DATA') {
            populateFields(data.firstName, data.lastName, data.emailAddress, data.summary, data.publicProfileUrl)
        }
    }

    function populateFields(first_name, last_name, email, summary, linkedin_public_profile_url) {

        last_name_field.val(last_name);

        linkedin_public_profile_url_field.val(linkedin_public_profile_url);
    }

              TITLE:                       Reporting Analyst
LOCATION:              Charleston, SC
STATUS:                  Full-time, exempt
Education:               Bachelor’s Degree preferred
 
SUMMARY:
Levelwing is seeking a highly-motivated individual with an analytical mindset to aid in the reporting of web traffic and performance. This individual must retain a high degree of curiosity and a passion for data analysis. The Reporting Analyst must be proficient in data aggregation and synthesis to power actionable findings and recommendations. This individual needs a positive “can do” mindset to meet a variety of business objectives.
 
DUTIES &amp; RESPONSIBILITIES:

Monitors and reports on a multitude of datasets including but not limited to: Website Performance, Marketing Channel Effectiveness, User Behavior and Engagement, Primary and Secondary Conversion Metrics
Comprehensive understanding of Web Analytics tools and knowledge of platforms that integrate with and impact digital data collection
Construct reports on a weekly, monthly, and ad-hoc basis
Identify and troubleshoot data collection issues and anomalies
Educate and support teammates through recommended reporting efficiencies
Present information in a consumable and impactful manner
Collaborate with Implementation Specialists, Media Analysts, and Web Developers on data collection methods
Create processes that streamline data collection and reporting needs
Ensure high level of accuracy and attention to detail in all work
Create and maintain data sourcing references and Analytics Design blueprints
Think creatively to analyze metrics from a variety of perspectives with the goal of providing valuable insight

Precise communications skills
Ability to learn new technology platforms in a timely manner
Complex dataset consumption
Comprehension of digital marketing channels and strategies
Strong attention to detail
Exceptional analytical skills
Ability to multi-task and simultaneously work in multiple work streams
Proficient on Microsoft Office Products
Google Analytics, Adobe Analytics, Google Data Studio, Power BI experience a plus

    function getApplyWithIndeedUrl() {
        var indeed_fragment = "apply_with_indeed=true";
        var current_url = window.location.href;

        if (current_url.indexOf(indeed_fragment) !== -1) {
            return current_url;
        }

        if (current_url.indexOf('?') !== -1) {
            return current_url + '&' + indeed_fragment;
        }

        return current_url + '?' + indeed_fragment;
    }

    (function handle_character_limit_on_paragraph() {
        var show_limit_message = function(e) {
            var $target = $(e.currentTarget);
            var max_character = Number($target.data('maxlength'));
            var current_length = $target.val().length;
            var remaining_length = max_character - current_length;
            var message_text = 'Not more than ' + max_character + ' characters';

            if(remaining_length  0) {
                message_text = remaining_length + ' characters remaining';
            }

            var message = $('', {
                class: 'js-max-length-message text-muted',
                text: ' ('+ message_text +')'
            });
            $target.siblings('.js-max-length-message').remove();
            $target.siblings('label').after(message);
        };

        var handle_keydown = function(e) {
            var $target = $(e.currentTarget);
            var max_character = Number($target.data('maxlength'));
            var current_length = $target.val().length;

            if ([8, 13, 27, 38, 40, 9, 39, 37].indexOf(e.keyCode) > -1) {
                return true;
            }

            if (current_length >= max_character) {
                return false;
            }
        }

        $('textarea[maxlength]')
            .each(function(idx, item){
                var $target = $(item);
                var maxlength = $target.attr('maxlength');
                $target.removeAttr('maxlength').data('maxlength', maxlength);
            })
                .on('keyup', show_limit_message)
                .on('keydown', handle_keydown)
                .on('input', show_limit_message)
                .trigger('keyup');
    })();
