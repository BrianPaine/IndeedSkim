        function RedirectToRP(url, showPopup) {

            var phenomRedirectURL = 'https://jobsbeta.microsoft.com/authenticate/social/sso';

            var authTypeId = '';

            var authTypeName = '';

            if (authTypeId) {

                authTypeName = 'Anonymous';

                if (authTypeId == 1 && authTypeName == 'Windows Live')

                    authTypeName = (authTypeName.split(" "))[1];

            }

            else

                authTypeName = '';

            if (showPopup) {

                var title = "Please note:";

                var message = "You’ve selected a job opening featured within the beta version of our new Global Careers site. You can review your currently selected job description and complete your application, however, you’ll want to return to our current site to search for additional jobs.For accessibility support, please contact Microsoft Disability Recruiting Team.";

                RedirectAlert(title, message, url, authTypeName, phenomRedirectURL);

            }

            else

            {

                var form = $('');

                form = form.append('');

                if (authTypeName != null && authTypeName != '')

                    authTypeName = authTypeName.toLowerCase();

                form = form.append('');

                form = form.appendTo('body').submit().remove();

            }

        }

    function getParameterByName(name) {

        name = name.replace(/[\[]/, "\\\[").replace(/[\]]/, "\\\]");

        var regexS = "[\\?&]" + name + "=([^&#]*)";

        var regex = new RegExp(regexS);

        var results = regex.exec(window.location.href);

        if (results == null)

            return "";

        else

            return decodeURIComponent(results[1].replace(/\+/g, " "));

    }

    function RemoveComma(validatorId) {

        var errorMsg = document.getElementById(validatorId);

        var tmp = errorMsg.innerHTML;

        var i = tmp.lastIndexOf(",");

        errorMsg.innerHTML = tmp.substr(0, i) + tmp.substr(i + 1, tmp.length);

    }

    function ChangeState(chkObj) {



        if (chkObj != null) {

            var declinetostate = "08" //"decline to state";

            var id = $(chkObj).parent().siblings("input:hidden").val();



            if (id == declinetostate) {

                $(".raceSelector").each(function (index) {

                    if ($(this).siblings("input:hidden").val() != id) {

                        $(this).children(":checkbox").attr("checked", false);

                    }

                }

                 );

            }

            else {

                $(".raceSelector").each(function (index) {

                    if ($(this).siblings("input:hidden").val() == declinetostate) {

                        $(this).children(":checkbox").attr("checked", false);



                    }

                }

                    );

            }

        }

    }
