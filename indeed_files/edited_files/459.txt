	function load_script(src){

		var h = document.getElementsByTagName('head')[0];

		var e = document.createElement('script');

		e.src = src;

		h.appendChild(e);

	}

	function load_link(href){

		var h = document.getElementsByTagName('head')[0];

		var e = document.createElement('link');

		e.rel = 'stylesheet';

		e.type = 'text/css';

		e.href = href;

		h.appendChild(e);
