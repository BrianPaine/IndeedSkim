    ( function ( document, window, index ) {
        if (isNotFlatNav()) {
            return;
        }

        var dHeight			= 0,
            wHeight			= 0,
            wScrollBefore	= 300,
            wScrollCurrent	= window.pageYOffset;
            wScrollDiff		= 0;

        var globalNavTop = 0;
        var globalNavBar = document.getElementsByClassName("flat-nav")[0];
        var cobrandBar = document.getElementsByTagName("cars-cobrand-bar");
        var bannerBar = document.getElementsByTagName("cars-mobile-app-banner");
        var carsLogo = document.getElementById("cars-com-logo");
        var navMenu = document.getElementsByClassName("global-nav__menu")[0];
        var angularProfileComponent = document.getElementsByTagName("cars-nav-user-settings");



        var preventDefault = function(event) {
            if (window.innerWidth  globalNavTop ? "0px" : (globalNavTop - wScrollCurrent) + "px");
        }

        function checkLogoAnimation() {
            if (wScrollCurrent  0) {
                globalNavTop = cobrandBar[0].getBoundingClientRect().height;
                adjustNavBarTop();

                cobrandBar[0].addEventListener("DOMSubtreeModified", function() {
                    globalNavTop = cobrandBar[0].getBoundingClientRect().height;
                    adjustNavBarTop();
                }, false);
            }
        }

        function addMobileBannerEvent() {
            if (bannerBar.length > 0) {
                globalNavTop = bannerBar[0].getBoundingClientRect().height;
                adjustNavBarTop();

                bannerBar[0].addEventListener("DOMSubtreeModified", function() {
                    globalNavTop = bannerBar[0].getBoundingClientRect().height;
                    adjustNavBarTop();
                }, false);
            }
        }

        function toggleMobileMenu() {
            var isChecked = document.getElementById("mobile-trigger-controller").checked;
            var body = document.getElementsByTagName("body")[0];

            if (isChecked) {
                body.classList.add('hide-overflow');
                window.addEventListener("touchmove", preventDefault, options);
                navMenu.style.top = (globalNavBar.getBoundingClientRect().top + globalNavBar.getBoundingClientRect().height) + "px";
            } else {
                body.classList.remove('hide-overflow');
                window.removeEventListener("touchmove", preventDefault, options);
            }
        }

        function addNavHamburgerEvent() {
            document.getElementById("global-nav_menu-trigger").addEventListener("click", function() {
                setTimeout(function() {
                    toggleMobileMenu();
                }, 0);
            });
        }

        function addProfileScrollLockFix() {
            if (angularProfileComponent.length > 0) {
                var profileToggleElement = angularProfileComponent[0].getElementsByClassName("cui-flyout");

                if (profileToggleElement.length > 0) {
                    profileToggleElement[0].addEventListener("click", function(event) {
                        var menuToggleElement = document.getElementById("mobile-trigger-controller");

                        if (menuToggleElement.checked) {
                            setTimeout(function() {
                                menuToggleElement.checked = false;
                                toggleMobileMenu();
                                // Re-add the hide-overflow class because we clicked the profile logo
                                document.getElementsByTagName("body")[0].classList.add('hide-overflow');
                            }, 0);
                        }
                    });
                }
            }
        }

        function addWindowScrollEvent() {
            window.addEventListener( 'scroll', function() {
                var body = document.body;
                var html = document.documentElement;
                dHeight = Math.max(body.scrollHeight, body.offsetHeight, html.clientHeight, html.scrollHeight, html.offsetHeight);
                wHeight = window.innerHeight;
                wScrollCurrent = window.pageYOffset;
                wScrollDiff = wScrollBefore - wScrollCurrent;

                checkLogoAnimation();

                // scrolled to the very top; element sticks to the top
                if( wScrollCurrent  0 && document.body.classList.contains('header--hidden') ) {
                    document.body.classList.remove('header--hidden');
                // scrolled down
                } else if( wScrollDiff = dHeight && document.body.classList.contains('header--hidden') ) {
                        document.body.classList.remove('header--hidden');
                    } else {
                        document.body.classList.add('header--hidden');
                    }
                }
                wScrollBefore = wScrollCurrent;

                adjustNavBarTop();
            });
        }
    }( document, window, 0 ));
