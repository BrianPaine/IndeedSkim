    function doHighlight(bodyText, searchTerm, color) {
        highlightStartTag = "";
        highlightEndTag = "";

        var newText = "";
        var i = -1;
        var lcSearchTerm = searchTerm.toLowerCase();
        var lcBodyText = bodyText.toLowerCase();
        while (bodyText.length > 0) {

            i = lcBodyText.indexOf(lcSearchTerm, i+1);
            if (i ", i) >= bodyText.lastIndexOf("<", i) ) {
                    var pattern = /^\W$/;

                    if(pattern.test(bodyText.charAt(i-1)) && pattern.test(bodyText.charAt(i+lcSearchTerm.length ))){
                        // skip anything inside a  block
                        if (lcBodyText.lastIndexOf("/script>", i) >= lcBodyText.lastIndexOf("<script", i)) {
                            newText += bodyText.substring(0, i) + highlightStartTag + bodyText.substr(i, searchTerm.length) + highlightEndTag;
                            bodyText = bodyText.substr(i + searchTerm.length);
                            lcBodyText = bodyText.toLowerCase();
                            i = -1;
                        }
                    }
                }
            }
        }
        return newText;
    }

    function highlightSearchTerms(searchText) {
        if(searchText != null){
            searchText = trim(searchText);
            searchArray = searchText.split(" ");
            contentObj = document.getElementById('highlight');
            var bodyText = contentObj.innerHTML;
            var j = 0;

            for (var i = 0; i < searchArray.length; i++) {
                if(j == 10){
                    j=0;
                }
                if(searchArray[i].length !=0) {
                    bodyText = doHighlight(bodyText, searchArray[i],KeyWordGetColor(j));
                    j++;
                }
            }

            contentObj.innerHTML = bodyText;
        }
        return false;
    }

    function KeyWordGetColor(idx) {
        var color = '#CCCCCC';

        switch (idx) {
            case 0:
                color='#FFFF00'; break;
            case 1:
                color='#00FFFF'; break;
            case 2:
                color='#87CEFA'; break;
            case 3:
                color='#ADFF2F'; break;
            case 4:
                color='#FF91FF'; break;
            case 5:
                color='#FFFAA0'; break;
            case 6:
                color='#CCFFCC'; break;
            case 7:
                color='#FFA07A'; break;
            case 8:
                color='#FFD00F';break;
            case 9:
                color='#A9A9A9';break;
            default:
                break;
        }

        return color;
    }

    function trim(s) {
        while (s.substring(0,1) == ' ') {
            s = s.substring(1,s.length);
        }

        while (s.substring(s.length-1,s.length) == ' ') {
            s = s.substring(0,s.length-1);
        }
        return s;
    }
