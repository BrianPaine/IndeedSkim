            $(document).ready(function(){
                $('#navBarMobile').slicknav({
                    label: ''
                });

                // Logo on on scroll down
                var didScroll;
                var lastScrollTop = 0;
                var delta = 5;
                var navbarHeight = $('.mainNavigation').outerHeight();

                $(window).scroll(function(event){
                    didScroll = true;
                });

                setInterval(function() {
                    if (didScroll) {
                        hasScrolled();
                        didScroll = false;
                    }
                }, 250);

                function hasScrolled() {
                    var st = $(this).scrollTop();
                    // Make sure they scroll more than delta
                    if(Math.abs(lastScrollTop - st)  lastScrollTop && st > navbarHeight){
                        // Scroll Down
                        $('.logo').removeClass('logoInBanner');
                        $('.mainNavigation').removeClass('removeLogo').addClass('addLogo');
                        $('.navBarLogoLink').show();
                    } else {
                        // Scroll Up Top
                        if(st <= 0) {
                            $('.mainNavigation').removeClass('addLogo').addClass('removeLogo');
                            $('.logo').addClass('logoInBanner');
                            $('.navBarLogoLink').hide();
                        }
                    }
                    
                    lastScrollTop = st;
                }

            });
